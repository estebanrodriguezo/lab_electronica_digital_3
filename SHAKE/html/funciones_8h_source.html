209static void xci_iota(uint64_t *state,int ir){
210 uint64_t s_copy[25] = {0}; // Creamos una copia temporal del estado
211 int i,j;
212 for(i=0;i<25;i++){
213 s_copy[i]=state[i]; // Copiamos el estado original a la copia temporal
214 }
215
216// Paso XOR (X) y Chi
217 for (i = 0; i < 5; i++) { // Bucle externo para recorrer las filas del estado.
218 for (j = 0; j < 5; j++) { // Bucle interno para recorrer las columnas del estado.
219 uint64_t A = s_copy[i + 5 * j]; // Obtenemos el valor de A del estado copiado.
220 uint64_t B = s_copy[(i + 1) % 5 + 5 * j]; // Obtenemos el valor de B del estado copiado.
221 //se realiza una operación XOR entre el elemento A, el elemento B y una máscara de bits 0xFFFFFFFFFFFFFFFFULL. Esto cambia algunos bits en el elemento B según el valor de C.
222 uint64_t C = s_copy[(i + 2) % 5 + 5 * j]; // Obtenemos el valor de C del estado copiado.
223
224 state[i + 5 * j] = A ^ ((B ^ 0xFFFFFFFFFFFFFFFFULL) & C); // Paso de Chi los bits de A se modifican según los valores de B y C.
225 // Realizamos la operación XOR (X) y Chi en el elemento actual del estado original.
226 }
227 }
228
229 // Paso Iota (I)
230 state[0] ^= RC[ir]; // Realizamos la operación Iota (I) en el primer elemento del estado original utilizando la constante RC correspondiente.
